Koneksi ke Google Drive

Hanya 4 langkah mudah — tanpa Cloud Console

Gunakan Google Apps Script sebagai jembatan antara website ini dan Google Drive Anda. Gratis, aman, dan tidak perlu pengaturan teknis yang rumit.

1
Buka Google Apps Script
Kunjungi script.google.com — pastikan sudah login dengan petarungmajene@gmail.com.
Klik tombol "+ New project" (pojok kiri atas). Hapus semua kode yang ada di editor (Ctrl+A lalu Delete).
2
Salin kode di bawah ini, tempel ke editor
Klik Salin Kode, lalu tempel (Ctrl+V) ke editor Apps Script:
Google Apps Script (Code.gs) 
function doGet(e){return respond(handle(e))}
function doPost(e){return respond(handle(e))}
function respond(d){
  var o=ContentService.createTextOutput(JSON.stringify(d));
  o.setMimeType(ContentService.MimeType.JSON);
  return o;
}
function handle(e){
  try{
    var a=e.parameter.action,b={};
    if(e.postData&&e.postData.contents){
      b=JSON.parse(e.postData.contents);
      if(!a)a=b.action;
    }
    if(a==='get')    return{ok:true,data:getData()};
    if(a==='save')   {saveData(b.data);return{ok:true};}
    if(a==='upload') return uploadFile(b);
    return{ok:false,error:'Unknown action: '+a};
  }catch(err){return{ok:false,error:err.toString()}}
}
function getFolder(){
  var n='Arsip Tata Ruang Majene';
  var it=DriveApp.getFoldersByName(n);
  return it.hasNext()?it.next():DriveApp.createFolder(n);
}
function getData(){
  var f=getFolder();
  var it=f.getFilesByName('arsip_data.json');
  if(it.hasNext())return JSON.parse(it.next().getBlob().getDataAsString());
  return{fpr:[],suket:[],skkr:[],surat:[]};
}
function saveData(d){
  var f=getFolder(),c=JSON.stringify(d);
  var it=f.getFilesByName('arsip_data.json');
  if(it.hasNext())it.next().setContent(c);
  else f.createFile('arsip_data.json',c,MimeType.PLAIN_TEXT);
}
function uploadFile(b){
  var folder=getFolder();
  if(b.subfolder){
    var sit=folder.getFoldersByName(b.subfolder);
    folder=sit.hasNext()?sit.next():folder.createFolder(b.subfolder);
  }
  var bytes=Utilities.base64Decode(b.data);
  var blob=Utilities.newBlob(bytes,b.mime||'application/octet-stream',b.name);
  var file=folder.createFile(blob);
  return{ok:true,id:file.getId(),name:b.name,url:'https://drive.google.com/file/d/'+file.getId()+'/view'};
}
3
Deploy sebagai Web App
Setelah menempel kode, tekan Ctrl+S untuk simpan.
Lalu klik menu Deploy → New deployment:
⚙️ Type: pilih Web app
👤 Execute as: Me
🌐 Who has access: Anyone
Klik Deploy → klik Authorize access → pilih akun petarungmajene@gmail.com → Allow → klik Done
